使用sea.js进行模块化加载

SeaJS是一个遵循CommonJS规范的JavaScript模块加载框架。其主要目的是解决目前JS编程中普遍存在的混乱依赖关系和代码纠缠问题，可以实现JS的模块化开发及按需加载，将工程师从繁重的JS文件及对象依赖处理中解放出来，从而专注于代码本身逻辑。

SeaJS本身仅有个位数的API，因此学习起来毫无压力。具体请参照SeaJS官方文档或官方主页：http://seajs.org/docs/。

CommonJS规范
根据CommonJS规范，一个单独的文件就是一个模块。加载模块使用require方法，该方法读取一个文件并执行，最后返回文件内部的exports对象。 
define(function(require, exports) {
    var c= require("./c");//此模块依赖c模块
    exports.a = function () {
        console.log("a是对外暴露接口")
    };
    exports.b = 'b';//b是对外暴露字符串
});


define(function(require, exports, module) {
    var c = require("./c");//此模块依赖c模块
    module.exports = {
        a:function(){
            console.log("a是对外暴露接口")
        },
        b:"b" //b是对外暴露字符串
    }
});

//以上两例导出结果均是
//{
//    a:function(){},
//    b:"b"
//}


//如不需要exports返回一个对象，只需要它返回一个函数。这时，就要写成module.exports


define(function(require, exports, module) {
    var c = require("./c");//此模块依赖c模块
    module.exports = function(){
        console.log("对外暴露function") //
    }
});
